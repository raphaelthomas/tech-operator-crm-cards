## 2️⃣ Single Operator CRM Pocket Card (Solo/On-Call)

```
\section{Card 2: Single Operator CRM (Solo Focus)}

\textbf{CRITICAL ACTIONS (HIGH-STRESS)}

\hline

\hline
\end{tabular}

\textbf{SBAR: ESCALATION FRAMEWORK}
\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{S}ituation & Immediate problem and impact. \\
\hline
\textbf{B}ackground & Context, history, and attempts made. \\
\hline
\textbf{A}ssessment & Your analysis, severity, and risk. \\
\hline
\textbf{R}ecommendation & Specific action needed from the recipient. \\
\hline
\end{tabular}

\subsection{Page 2: Structured Problem Solving}

\textbf{FOR-DEC: DECISION FRAMEWORK}
\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{F}acts & Gather current system data. Ignore assumptions. \\
\hline
\textbf{O}ptions & List possible solutions. \\
\hline
\textbf{R}isks & Assess impact of action vs. no action. \\
\hline
\textbf{D}ecision & Select the best course of action. \\
\hline
\textbf{E}xecution & Execute deliberately. Verbalize command first. \\
\hline
\textbf{C}heck & Verify result. If failed, return to \textbf{Facts}. \\
\hline
\end{tabular}

\vspace{0.5cm}

\textbf{ERROR PREVENTION PROTOCOL}
\begin{itemize}
    \item \textbf{3-WAY Rule:} For irreversible commands: \textbf{Say It} $\to$ \textbf{Type It} $\to$ \textbf{Review It} (read back).
    \item \textbf{Documentation:} Live-document all critical actions and commands immediately.
    \item \textbf{Checklist Discipline:} Use the official Runbook/Checklist; do not rely on memory.
\end{itemize}

## 3️⃣ Deployment Safety Card (Change Management)

```
\section{Card 3: Deployment Safety (Change Focus)}
\subsection{Page 1: Planning \& Verification}

\textbf{STEP 1: BRIEFING - COMMIT \& CLARIFY (AIM-C)}
\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{A}im/\textbf{I}ntent & State Goal \& Review Main Plan. \\
\hline
\textbf{M}ishaps/\textbf{C}ontingency & Identify 2-3 risks and define **Contingency Triggers** (Plan B). \\
\hline
\textbf{Close the Loop} & \textbf{Mandate Confirmation}: Ask key team members to verbally repeat the Triggers/Risks. \\
\hline
\end{tabular}

\vspace{0.5cm}

\textbf{STEP 2: VERIFICATION - CHECKLIST \& GO/NO-GO}
\begin{itemize}
    \item \textbf{Rollback Path}: Is the rollback procedure validated and ready \textbf{immediately}?
    \item \textbf{System Capacity}: Is resource usage \textbf{below 60\%} and stable?
    \item \textbf{Monitoring}: Is the dashboard verified to be displaying \textbf{live, accurate} data?
    \item \textbf{Go/No-Go Call}: All key stakeholders must give a \textbf{verbal "GO"}.
\end{itemize}

\subsection{Page 2: Execution \& Failure Protocol}

\textbf{STEP 3: EXECUTION DISCIPLINE}
\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{Time Sync} & \textbf{Start a common, visible timer} now. \\
\hline
\textbf{Sterile Cockpit} & Restrict comms to essential execution/status updates only. \\
\hline
\textbf{Two-Person Rule} & Designate \textbf{Operator} and \textbf{Monitor}. Monitor verifies command and metrics. \\
\hline
\textbf{Stop-Point Checks} & Define \textbf{3-5 objective checks} that must be \textbf{GREEN} before proceeding. \\
\hline
\end{tabular}

\vspace{0.5cm}

\textbf{STEP 4: IMMEDIATE FAILURE PROTOCOL}
\begin{itemize}
    \item \textbf{The 5-Minute Rule}: If system not green by \textbf{Contingency Trigger Time}, \textbf{initiate ROLLBACK immediately}.
    \item \textbf{Rollback Action}: Announce \textbf{"ROLLING BACK NOW!"} $\to$ Capture state $\to$ Execute rollback.
    \item \textbf{Verification}: Do not declare success until rollback is \textbf{complete} and system is stable.
\end{itemize}

\hline

\textbf{TRIAL-RUN} & \textbf{T}est $\to$ \textbf{R}eplicate $\to$ \textbf{I}solate $\to$ \textbf{A}nalyze $\to$ \textbf{L}og $\to$ \textbf{R}epeat $\to$ \textbf{U}pdate $\to$ \textbf{N}otify. \\
\hline
\textbf{5 Whys} & Ask \textbf{"Why?"} repeatedly to move past symptom to next dependency/team. \\
\hline
\end{tabular}

\vspace{0.5cm}

\textbf{PHASE 2: INCIDENT \& RESOURCE LIMITS}
\begin{itemize}
    \item \textbf{The 30-Minute Rule}: If cause is \textbf{not found within 30 minutes}, \textbf{STOP} and \textbf{ESCALATE} immediately.
    \item \textbf{Context Protocol}: Before troubleshooting, gather \textbf{User ID}, \textbf{Exact Time}, \textbf{Error Message}, and \textbf{Last Action}.
    \item \textbf{Documentation}: Live-document commands, logs, and hypotheses in the ticket \textbf{as they happen}.
\end{itemize}

\subsection{Page 2: Communication \& Handoff}

\textbf{PHASE 3: CUSTOMER COMMUNICATION (SBAR-C)}
\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{S}ituation & What is the current problem and impact? \\
\hline
\textbf{B}ackground & Steps taken and relevant history. \\
\hline
\textbf{A}ssessment & Your diagnosis and severity level. \\
\hline
\textbf{R}ecommendation & Proposed fix or needed action. \\
\hline
\textbf{C}losing & Mandate next status update time. (\textbf{Always deliver on time.}) \\
\hline
\end{tabular}

\vspace{0.5cm}

\textbf{PHASE 4: ESCALATION HANDOFF (Internal)}
\begin{itemize}
    \item \textbf{SME Handoff}: Use \textbf{SBAR} (without 'C') to brief the Subject Matter Expert.
    \item \textbf{Final Check}: Before declaring success, verify the fix addresses the \textbf{root cause}, not just the symptom.
\end{itemize}
