\cardsection{1. FOR-DEC Decision Loop}

\subsectiontitle{THE SIX-STEP COMMAND STRUCTURE}
\begin{tabularx}{\linewidth}{|p{0.9cm}|X|}
\hline
\textbf{F}acts & \textbf{Data only.} What is the system state? What is the user seeing? \\
\hline
\textbf{O}ptions & Brainstorm all possible solutions (including No-Action). \\
\hline
\textbf{R}isks & Assess risk of each option. Prioritize actions with safe failure modes. \\
\hline
\textbf{D}ecision & State the selected course of action clearly. \\
\hline
\textbf{E}xecution & Initiate the procedure (slow, deliberate commands). \\
\hline
\textbf{Check} & Verify the results. If failure, return to Facts (F). \\
\hline
\end{tabularx}

\cardsection{2. The 3-Second Rule}

\subsectiontitle{MANDATE FOR ACTION}
\begin{itemize}
    \item If you are \textbf{thinking} about calling for help, \textbf{call now.}
    \item If you are \textbf{thinking} about initiating rollback, \textbf{do it now.}
    \item If you are \textbf{thinking} about asking for a break, \textbf{ask now.}
\end{itemize}

\cardsection{3. Handover Protocol}

\subsectiontitle{WHEN SHIFTING COMMAND/CONTROL}
\begin{itemize}
    \item \textbf{Situation}: State current time, status, and what is stable.
    \item \textbf{Target}: What is the primary metric the new owner should monitor?
    \item \textbf{Action}: What is the current plan/command being executed?
    \item \textbf{Request}: What is the next command to be executed by the new owner?
\end{itemize}
