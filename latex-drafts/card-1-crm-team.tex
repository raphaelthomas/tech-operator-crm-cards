\cardsection{1. The Four CRM Pillars}

\subsectiontitle{SHARED SITUATIONAL AWARENESS (S.A.)}
\begin{tabularx}{\linewidth}{|p{0.8cm}|X|}
\hline
\textbf{Probe} & Pause, step back. "What do we know, what don't we know?" \\
\hline
\textbf{Alert} & State the problem and perceived risk clearly: "I am concerned that..." \\
\hline
\textbf{Challenge} & Offer an alternative: "I disagree. Let's try X, not Y, because..." \\
\hline
\textbf{Stop} & Mandatory, non-negotiable stop: "STOP THE LINE. Break." \\
\hline
\end{tabularx}

\cardsection{2. Closed-Loop Communication}

\subsectiontitle{CRITICAL INFORMATION TRANSFER}
\begin{itemize}
    \item \textbf{Sender}: Clearly state the action or instruction ("Deploy rollback script V3.1").
    \item \textbf{Receiver}: Repeat the instruction exactly ("Rollback script V3.1 confirmed").
    \item \textbf{Sender}: Confirm understanding and correctness ("Confirmed V3.1 is correct").
    \item \textbf{Completion}: Receiver states outcome ("V3.1 deployed. System health is GREEN.").
\end{itemize}

\cardsection{3. Peer Intervention (IMSAFE)}

\subsectiontitle{CHECK YOUR TEAMMATE'S READINESS}
\begin{itemize}
    \item \textbf{I}llness: Are they physically well?
    \item \textbf{M}edication: Any substances affecting focus?
    \item \textbf{S}tress: Are they overloaded or under duress?
    \item \textbf{A}lcohol: Obviously, none.
    \item \textbf{F}atigue: Have they had adequate rest?
    \item \textbf{E}motion: Are they clearly angry or distressed?

    \vspace{0.1cm}
    \textit{If any check is \textbf{Red}, mandate a break or handoff.}
\end{itemize}
