\section{Team Principles \& Readiness}

\subsection{CORE PILLARS}

\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{1. Situational Awareness (SA)} & \textbf{10 for 10 Check:} Pause every 10 mins: Problem? Facts? Team? Plan! Distribute! \\
\hline
\textbf{2. Communication} & \textbf{Close the Loop:} Sender/Receiver confirmation: \textbf{Said $\to$ Heard $\to$ Understood $\to$ Done}. \\
\hline
\textbf{3. Assertiveness} & \textbf{PACE Ladder:} \textbf{P}robe $\to$ \textbf{A}lert $\to$ \textbf{C}hallenge $\to$ \textbf{E}mergency (Act to Stop). \\
\hline
\textbf{4. Handoffs} & \textbf{BAT:} \textbf{B}ackground, \textbf{A}ction (Current State), \textbf{T}asks (Future Plan). \\
\hline
\end{tabular}

\subsection{PERSONAL READINESS CHECK}

\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{IMSAFE} & \textbf{I}llness, \textbf{M}edication, \textbf{S}tress, \textbf{A}lcohol/\textbf{F}atigue, \textbf{E}motion. \\
\hline
\textbf{Mandate} & If any check is \textbf{Red}, demand a break or handoff immediately. \\
\hline
\end{tabular}

\section{Action Tools \& Error Management}

\subsection{CRITICAL ACTION FRAMEWORKS}

\begin{tabular}{|l|p{6cm}|}
\hline
\textbf{FOR-DEC} & \textbf{F}acts, \textbf{O}ptions, \textbf{R}isks, \textbf{D}ecision, \textbf{E}xecution, \textbf{C}heck. \\
\hline
\textbf{Sterile Cockpit} & \textbf{Restrict all conversation} to only essential tasks during critical phases (e.g., major restarts). \\
\hline
\textbf{3-WAY Rule} & Critical Commands: \textbf{Say It} $\to$ \textbf{Point It} $\to$ \textbf{Do It} (After Monitor confirms). \\
\hline
\end{tabular}

\subsection{ERROR PREVENTION PROTOCOL}
\begin{itemize}
    \item \textbf{Double Check:} Apply Two-Person Rule for any irreversible change.
    \item \textbf{The 8-Hour Rule:} Mandatory break/handoff after 8 hours of critical work.
    \item \textbf{Documentation Mandate:} Live-document commands, actions, and outcomes \textbf{as they happen}.
\end{itemize}
